\section{$Ab$ $initio$ alloy theory}
\subsection{Density functional theory }
\begin{frame}
\frametitle{Density functional theory (DFT)}
%\begin{columns}[c]
%\column{2in}
\begin{itemize}
  \item the ground state of an interacting electron system is uniquely described by an energy functional of the electron density,
  \item the true ground state electron density minimizes the energy functional and the minimum gives the total energy.
\end{itemize}
\begin{equation}
[-\nabla^2+V_{eff}(\textbf{r})]\psi_i^k = \epsilon_i^k \psi_i^k,
\label{ks}
\end{equation}
\begin{equation}
V_{eff}(\textbf{r})= V_{ext}(\textbf{r}) +\int \frac {n(\textbf{r}')}{|\textbf{r}-\textbf{r}'|} d \textbf{r}' +\mu_{xc}(\textbf{r}),
\end{equation}
\begin{equation}
\mu_{xc}\equiv\delta E_{xc}[n(\textbf{r})]/\delta n(\textbf{r}).
\end{equation}
\begin{equation}
E_{tot}=T_s+\int v_{ext}(\textbf{r})n(\textbf{r})d^3r+\frac 1 2 \int\int\frac{n(\textbf{r})n(\textbf{r}')}{|\textbf{r}-\textbf{r}'|}d^3rd^3r'+E_{xc}
\label{dft}
\end{equation}
